# Sandy daily refresh cron job
# Runs at 6:00 AM UTC every day (before most MLB games start)
#
# Install with:
#   crontab sandy/scripts/crontab.txt
#
# Or append to existing crontab:
#   (crontab -l 2>/dev/null; cat sandy/scripts/crontab.txt) | crontab -
#
# Ensure these env vars are set in ~/.sandy_env or the script's environment:
#   MLB_DB_HOST, MLB_DB_PORT, MLB_DB_NAME, MLB_DB_USER, MLB_DB_PASSWORD
#   TELEGRAM_BOT_TOKEN, TELEGRAM_CHAT_ID

# Create log directory if needed
SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin:/home/ec2-user/.local/bin

# Daily refresh at 6 AM UTC
0 6 * * * /home/ec2-user/sandy/scripts/daily_refresh.sh >> /home/ec2-user/sandy/logs/cron.log 2>&1
